<insert id="insertMember"
		parameterType="com.kh.mybatis.member.Member">
		insert into member values(
		#{id},#{password},#{name},#{age},
		#{gender},#{phone},#{address},#{email},
		#{
		hobby,typeHandler =
		com.kh.mybatis.member.common.MemberTypeHandler
		},
		default
		)
</insert>
